1．メインフォームに、cFeatureRichと、その上に cRichPane を造る。
2．cRichPaneの名前は、Resourceとする
3．メインフォームに、以下の処理を追記


public partial class FormMain : Form
{
	/// <summary>
	/// フォームの位置・サイズをレジストリに記憶する
	/// </summary>
	private uFormShapePersister _formShapePersister;

	/// <summary>
	/// コンストラクタ
	/// </summary>
	public FormMain()
	{
		InitializeComponent();
		_formShapePersister = new uFormShapePersister(this);
	}

	/// <summary>
	/// 初期化処理
	/// </summary>
	/// <param name="sender"></param>
	/// <param name="e"></param>
	private void FormMain_Load(object sender, EventArgs e)
	{
		// フィーチャーの準備
		mainFeature.IsDrawEmptyBackground = false;
		richPaneResource.IdColor = Color.FromArgb(48, 48, 48);
		richPaneToolbar.IdColor = Color.FromArgb(48, 48, 48);
		uMes.SetDefault();	// この FeatureLoaderの前のこの位置で実行が必要

		// フィーチャーアーキテクチャの初期化を行う（言語設定より前に行うこと）
		uFeatureLoader2.SetResources(Properties.Resources.ResourceManager);
		uFeatureLoader2.SetUsingClass(GetType());
		mainFeature.Initialize(typeof(uFeatureLoader2));

		uMes.Current.ResetText(this);
		uTime.SetDayStrings(uMes.Current);

		// パラメータの処理
		mainFeature.GetFeatureRoot().ParseCommandLineParameter(Environment.GetCommandLineArgs());

		// 起動直後トークンを回しておく
		mainFeature.GetFeatureRoot().FlushFeatureTriggers();

	}
}
